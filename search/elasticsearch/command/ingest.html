            # 网络配置抓取处理

            1.将 device_type 属性内容改成小写 2.去掉属性 command 中的 .txt 3.将属性 device_type 中为 cisco_switch 的内容改为 catalyst

            PUT _ingest/pipeline/netconfig-pipline
{
  "processors": [
    {
      "lowercase": {
        "field": "device_type"
      }
    },
    {
      "gsub": {
        "field": "command",
        "pattern": "\\.txt",
        "replacement": ""
      }
    },
    {
      "script": {
        "lang": "painless",
        "source": """
        if(ctx.device_type.equals("cisco_switch")){
          ctx.device_type= "catalyst"
        }
        """
      }
    }
  ]
}

            # 索引自动添加时间戳

            PUT _ingest/pipeline/my_timestamp_pipeline
{
  "description": "Adds a field to a document with the time of ingestion",
  "processors": [
    {
      "set": {
        "field": "ingest_timestamp",
        "value": "{{_ingest.timestamp}}"
      }
    }
  ]
}


PUT my-index-timestamp
{
  "settings": {
    "index.default_pipeline": "my_timestamp_pipeline"
  }, 
  "mappings": {
    "properties": {
      "content": {
        "type": "text"
      },
      "ingest_timestamp": {
        "type": "date",
        "format": "yyyy-MM-dd'T'HH:mm:ss.SSSSSSSSSX"
      }
    }
  }
}


POST my-index-timestamp/_doc
{
  "content": "it is cool!"
}

            # 处理 MySQL 审计日志

            POST _ingest/pipeline/_simulate
{
  "pipeline": {
    "processors": [
      //取最后一个数字
      {
        "grok": {
          "field": "message",
          "patterns": [".*%{NUMBER:retcode}"]
        }
      },
      //将 message 按照逗号分隔，生成临时字段 tmp_message
      {
        "split": {
          "field": "message",
          "target_field": "tmp_message",
          "separator": ","
        }
      },
      //在临时字段数组中获取值
      {
        "script": {
          "source": "ctx.servertime = ctx.tmp_message[0]"
        }
      },
      {
        "script": {
          "source": "ctx.hostname = ctx.tmp_message[1]"
        }
      },
      {
        "script": {
          "source": "ctx.dbuser = ctx.tmp_message[2]"
        }
      },
      {
        "script": {
          "source": "ctx.dbhost = ctx.tmp_message[3]"
        }
      },
      {
        "script": {
          "source": "ctx.connectionid = ctx.tmp_message[4]"
        }
      },
      {
        "script": {
          "source": "ctx.queryid = ctx.tmp_message[5]"
        }
      },
      {
        "script": {
          "source": "ctx.operation = ctx.tmp_message[6]"
        }
      },
      {
        "script": {
          "source": "ctx.database = ctx.tmp_message[7]"
        }
      },
      //修改 servertime 字段日期格式 20210927 13:09:18 -> 2021-09-27T13:09:18
      {
        "script": {
          "source": """
            String[] str = ctx.servertime.splitOnToken(" ");
            String year = str[0].substring(0,4);
            String month = str[0].substring(4,6);
            String day = str[0].substring(6);
            ctx.servertime = year + "-" + month + "-" + day + "T" + str[1]
          """
        }
      },
      //删除临时字段
      {
        "remove": {
          "field": "tmp_message"
        }
      }
    ]
  },
  "docs": [
    {
      "_source": {
        "message": "20210927 13:09:18,acp_sas_node03,universe_op,127.0.0.1,7571784,0,DISCONNECT,,,0"
      }
    }
    ]
}

        "_index" : "_index",
        "_type" : "_doc",
        "_id" : "_id",
        "_source" : {

          "dbuser" : "universe_op",
          "message" : "20210927 13:09:18,acp_sas_node03,universe_op,127.0.0.1,7571784,0,DISCONNECT,,,0",

            # 处理网络 Syslog 日志

            PUT _ingest/pipeline/network-pipeline
{
  "processors": [
    // 根据 ip 地址为网络设备添加 device_name(设备名), device_type(设备类型), device_model(设备型号), device_software(软件版本), environment(运行环境) 字段
    {
      "enrich": {
        "policy_name": "network-enrich-policy",
        "field": "host",
        "target_field": "enrich-data",
        "ignore_failure": true
      }
    },
    // 移除 logstash grok 匹配异常时产生的字段
    {
      "remove": {
        "field": "tags",
        "ignore_failure": true
      }
    },
    // 删除 F5 健康检查产生的文档
    {
      "drop": {
        "if": "ctx.message == 'default send string'"
      }
    },
    // 处理 enrich 产生的前缀
    {
      "set": {
        "field": "device_model",
        "value": "{{enrich-data.device_model}}"
      }
    },
    {
      "set": {
        "field": "device_name",
        "value": "{{enrich-data.device_name}}"
      }
    },
    {
      "set": {
        "field": "device_software",
        "value": "{{enrich-data.device_software}}"
      }
    },
    {
      "set": {
        "field": "device_type",
        "value": "{{enrich-data.device_type}}"
      }
    },
    {
      "set": {
        "field": "environment",
        "value": "{{enrich-data.environment}}"
      }
    },
    {
      "remove": {
        "field": "enrich-data"
      }
    },
    // F5 提取日志等级
    {
      "grok": {
        "field": "message",
        "patterns": [
          "%{MONTH} %{NUMBER} %{TIME} %{NOTSPACE} %{WORD:level}"
        ],
        "if": "ctx.device_type == 'f5'",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    // topsec 提取日志等级
    {
      "grok": {
        "field": "message",
        "patterns": [
          "pri=%{NUMBER:level}"
        ],
        "if": "ctx.device_type == 'topsec'",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    // huawei 提取日志等级
    {
      "grok": {
        "field": "message",
        "patterns": [
          "/%{NUMBER:level}/"
        ],
        "if": "ctx.device_type.contains('huawei')",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    // nexus 提取日志等级
    {
      "grok": {
        "field": "message",
        "patterns": [
          "%{DEVICE}|%{FEX}"
        ],
        "pattern_definitions": {
          "FEX": "%%{WORD}-%{WORD}-%{HOUR:level}-%{WORD}",
          "DEVICE": "%%{WORD}-%{HOUR:level}-%{WORD}"
        },
        "if": "ctx.device_type == 'nexus'",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    // catalyst 提取日志等级
    {
      "grok": {
        "field": "message",
        "patterns": [
          "%{MONTH} %{NUMBER} %{TIME}: %%{WORD}-%{HOUR:level}"
        ],
        "if": "ctx.device_type == 'catalyst'",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    // fortigate 提取日志等级
    {
      "grok": {
        "field": "message",
        "patterns": [
          "level=\"%{WORD:level}\""
        ],
        "if": "ctx.device_type == 'fortigate'",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    // f5 日志告警等级统一处理
    {
      "script": {
        "source": """
           if(ctx.level == 'emerg'){
             ctx.level = 'emergency'
           }
           if(ctx.level == 'crit'){
             ctx.level = 'critical'
           }
           if(ctx.level == 'err'){
             ctx.level = 'error'
           }
           if(ctx.level == 'info'){
             ctx.level = 'information'
           }
        """,
        "if": "ctx.device_type == 'f5'"
      }
    },
    // huawei, nexus, catalyst, topsec 日志等级是 0~7, 统一处理 debug ~ emergency
    {
      "script": {
        "if": "ctx.device_type.contains('huawei') || ctx.device_type == 'nexus' || ctx.device_type == 'catalyst'  || ctx.device_type == 'topsec'",
        "source": """
          if(ctx.level == '0'){
            ctx.level = 'emergency'
          }
          if(ctx.level == '1'){
            ctx.level = 'alert'
          }
          if(ctx.level == '2'){
            ctx.level = 'critical'
          }
          if(ctx.level == '3'){
            ctx.level = 'error'
          }
          if(ctx.level == '4'){
            ctx.level = 'warning'
          }
          if(ctx.level == '5'){
            ctx.level = 'notice'
          }
          if(ctx.level == '6'){
            ctx.level = 'information'
          }
          if(ctx.level == '7'){
            ctx.level = 'debug'
          }
        """
      }
    },
    // npm 日志处理
    {
      "grok": { // 去掉开头的 <29>
        "field": "message",
        "patterns": [">%{GREEDYDATA:message}"],
        "if": "ctx.device_type == 'npm'",
        "on_failure": [
          {
            "set": {
              "field": "exception",
              "value": "grok"
            }
          }
        ]
      }
    },
    {
      "csv": {
        "field": "message",
        "target_fields": [
          "alert_time",
          "view_type",
          "area_disp_name",
          "group_disp_name",
          "source",
          "type",
          "event",
          "value",
          "thresholds",
          "start_time",
          "duration",
          "status",
          "level"
        ],
        "separator": "|",
        "if": "ctx.device_type == 'npm'"
      }
    },
    {
      "lowercase": {
        "field": "level",
        "if": "ctx.device_type == 'npm'"
      }
    },
    // npm 告警状态处理，开启和关闭
    {
      "script": {
        "source": """
             if(ctx.status == '已恢复'){
               ctx.status = 'close'
             }else {
               ctx.status = 'open'
             }
              String text = ctx.source + ctx.type + ctx.event + ctx.start_time; 
              ctx.sn =  text.encodeBase64().substring(0,11); // 通过组合字段 base64 加密后生成唯一的 sn, 用于告警中心判断关闭告警
          """,
        "if": "ctx.device_type == 'npm'"
      }
    }
  ]
}
