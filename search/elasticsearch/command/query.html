                # Elasticsearch 查询语句

                # 当 total value 大于10000时，查询完整数量

                #URI Search
GET shakespeare/_search?track_total_hits=true

#Query DSL
GET shakespeare/_search
{
   "query": {
     "match_all": {}
   }, 
   "track_total_hits": true
}

#加上size:{number}参数指定内容条数

                # 搜索 date 类型数据指定时区

                GET vba-app-logstash/_search
{
  "_source": "@timestamp",
  "query": {
    "range": {
      "@timestamp": {
        "gte": "2021-03-31T08:53:19",
        "lte": "2021-03-31T09:53:19",
        "time_zone": "Asia/Shanghai"
      }
    }
  }, 
  "sort": [
    {
      "@timestamp": {
        "order": "desc"
      }
    }
  ]
}

                也可以指定 UTC + 8 换算成中国时区：

                GET vba-app-logstash/_search
{
  "_source": "@timestamp",
  "query": {
    "range": {
      "@timestamp": {
        "gte": "2021-03-31T08:53:19+08:00",
        "lte": "2021-03-31T09:53:19+08:00"
      }
    }
  }, 
  "sort": [
    {
      "@timestamp": {
        "order": "desc"
      }
    }
  ]
}

                # 分桶后根据 doc_count 排序

                GET vba-app-logstash/_search
{
  "query": {
    "range": {
      "@timestamp": {
        "gte": "2021-03-31",
        "lte": "2021-03-31"
      }
    }
  },
  "size": 0,
  "aggs": {
    "time_bucket": {
      "date_histogram": {
        "field": "@timestamp",
        "interval": "minute",
        "order": {
          "_count": "desc"  #排序字段是doc_count，顺序是降序
        }
      }
    }
  }
}

                查询 IP 为 116.236.154.162 发送的请求路径的基数：

                GET *nginx-access*/_search
{
  "size": 0,
  "query": {
    "term": {
      "x-forwarded-for.keyword": {
        "value": "\"116.236.154.162\""
      }
    }
  },
  "aggs": {
    "request_count": {
      "cardinality": {
        "field": "request.keyword"  #查询有多种不同的request请求

                # 计算按照源目 IP 地址之前流量大小排序

                GET netflow-2021.12.03/_search?filter_path=aggregations
{
  "query": {
    "bool": {
      "filter": [
        {
          "range": {
            "@timestamp": {
              "gte": "2021-12-03T14:50:30",
              "lte": "2021-12-03T14:57:30",
              "time_zone": "Asia/Shanghai"
            }
          }
        }
      ]
    }
  }, 
  "aggs": {
    "ip": {
      "multi_terms": {
        "terms": [{
          "field": "netflow.ipv4_src_addr.keyword"
        }, {
          "field": "netflow.ipv4_dst_addr.keyword"
        }],
        "order": {
          "sum_bytes": "desc"
        }
      },
      "aggs": {
        "sum_bytes": {
          "sum": {
            "field": "netflow.in_bytes"
          }
        }
      }
    }
  }
}
