        # 使用 Kubernetes API 访问集群

        # 创建 serviceaccount 并绑定权限

        创建 serviceaccount apiserver-admin 并绑定默认集群角色 cluster-admin。

        kubectl create serviceaccount apiserver-admin
kubectl create clusterrolebinding apiserver-admin-cluster-admin --clusterrole='cluster-admin' --serviceaccount=default:apiserver-admin

        获取seviceaccount apiserver-admin 的 secret token：

        TOKEN=$(kubectl get secrets -o jsonpath="{.items[?(@.metadata.annotations['kubernetes\.io/service-account\.name']=='apiserver-admin')].data.token}"|base64 -d)

echo $TOKEN

        # 获取 Pod 资源

        Kubernetes 相关资源对象 Rest API 文档：https://kubernetes.io/docs/reference/kubernetes-api/

        获取 elastalert 命名空间下的所有 Pod。

        curl https://11.16.0.153:6443/api/v1/namespaces/elastalert/pods \
--header "Authorization: Bearer $TOKEN" --insecure -s \
| jq '.items[].metadata.name'

        获取指定 Pod 的运行状态。

        curl https://11.16.0.153:6443/api/v1/namespaces/elastalert/pods/elastalert-1/status \
--header "Authorization: Bearer $TOKEN" --insecure -s \
| jq '.status.phase'

# 返回结果
"Running"

        获取指定 Pod 的启动时间。

        curl https://11.16.0.153:6443/api/v1/namespaces/elastalert/pods/elastalert-1/status \
--header "Authorization: Bearer $TOKEN" --insecure -s \
| jq '.status.startTime'
